\documentclass{article}

% ----- page size -----
\usepackage[paperheight=5.5cm, paperwidth=5.5cm]{geometry}

% ----- fonts -----
\usepackage{fontspec}
\newcommand{\mainfont}{\fontspec{Helvetica Neue}}
\newcommand{\subfont}{\fontspec{Helvetica Neue Thin}}

% ----- colors -----
\usepackage{xcolor}
% paleturquoise (RGB: 143, 229, 216; CMYK: 38, 0, 6, 10)
\definecolor{color1}{RGB}{143, 229, 216}
% symphony blue (RGB: 24, 87, 227; CMYK: 92, 63, 0, 0)
\definecolor{color2}{RGB}{24, 87, 227}
% obaku (RGB: 255, 242, 127; CMYK: 1, 5, 62, 0)
\definecolor{color3}{RGB}{255, 242, 127}

% ----- tikz -----
\usepackage{tikz}
\usetikzlibrary{shapes, positioning}

\pgfdeclarelayer{bg}
\pgfsetlayers{bg,main}

\tikzstyle{octagon} = [
  shape = regular polygon,
  regular polygon sides = 8
]
\tikzstyle{star} = [
  shape = star,
  star point ratio = 1.3,
  star points = 8
]

% ----- logo itself -----

\begin{document}

\thispagestyle{empty}

\centering
\null\vfill

\begin{tikzpicture}
\node [circle, minimum width = 1cm, fill = color1] (n1) {};
\node [octagon, minimum width = 1cm, fill = color2, right = 0.4cm of n1] (n2) {};
\node [star, minimum width = 1cm, fill = color3, right = 0.4cm of n2] (n3) {};
\begin{pgfonlayer}{bg}
\draw [ultra thick, shorten <= -0.2cm, shorten >= -0.2cm] (n1) -- (n3);
\end{pgfonlayer}
\end{tikzpicture}

\vspace{0.25cm}

\sffamily
{\Huge\mainfont llmk} \\
